import { useState } from "react";

export function Counter({ children }) {
    const [count, setCount] = useState(0);
    return (
        <div>
            <button onClick={() => setCount(count + 1)}>+</button>
            <button onClick={() => setCount(count - 1)}>-</button>
            <div>{children(count)}</div>
        </div>
    );
}

<Counter>{(count) => <div>{count}</div>}</Counter>
